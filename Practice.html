// Java Version
class Stack {
	int[] items = new int[s];
	int top;
}

class IntStackDemo {
	static void push (int i, Stack s) {
		s.items[s.top++] = i;	
	}
}

static int pop(Stack s) {
	s.items[--s.top];
}

// RESOLVE Version

Facility Stack;
	Type Stack =
		Record 
			top: Integer;
			Stack: Array 0..4 of Integer;
		end;
	end;	
	
	operation Push (preserves val: Integer;  alters s: Stack);
	Procedure
		s.Stack[s.top] : = val;
		s.top := s.top + 1;
	end;
	
	operation Pop(alters s:Stack): Integer;
	Procedure
		s.top := s.top - 1;
		Pop := s.Stack[s.top];
	end;
end;
